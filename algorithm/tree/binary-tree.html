        return result

class Solution:
    def flatten(self, root: TreeNode) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        r = root
        while r:
            if r.left:
                p = r.left
                while p and p.right:
                    p = p.right
                p.right = r.right
                r.right = r.left
                r.left = None
            r = r.right
